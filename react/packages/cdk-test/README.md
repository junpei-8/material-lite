## Install

```bash
npm install --save cdk
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'cdk'
import 'cdk/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
